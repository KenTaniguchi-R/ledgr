// src/lib/demo-mode.ts
import { eq } from "drizzle-orm";
import { db as defaultDb, type LedgrDb } from "@/db";
import { userSettings } from "@/db/schema";

export const DEMO_HOUSEHOLD_ID = "00000000-0000-0000-0000-000000000000";

export function isDemoMode(userId: string, db: LedgrDb = defaultDb): boolean {
  const row = db
    .select({ demoMode: userSettings.demoMode })
    .from(userSettings)
    .where(eq(userSettings.userId, userId))
    .get();
  return row?.demoMode === true;
}

export function guardDemoMode(
  userId: string,
  db: LedgrDb = defaultDb,
): { error: string } | null {
  if (isDemoMode(userId, db)) {
    return { error: "Demo mode is read-only. Switch to your account to make changes." };
  }
  return null;
}
```

export async function toggleDemoMode(): Promise<{ success: true } | { error: string }> {
  const session = await getSession();
  if (!session) return { error: "Not authenticated" };

  const existing = db
    .select({ id: userSettings.id, demoMode: userSettings.demoMode })
    .from(userSettings)
    .where(eq(userSettings.userId, session.user.id))
    .get();

  if (existing) {
    db.update(userSettings)
      .set({ demoMode: !existing.demoMode, updatedAt: new Date().toISOString() })
      .where(eq(userSettings.id, existing.id))
      .run();
  } else {
    db.insert(userSettings)
      .values({ id: uuid(), userId: session.user.id, demoMode: true })
      .run();
  }

  revalidatePath("/", "layout");
  return { success: true };
}
```

function seedDemoTransactions(tx: any, householdId: string) {
  const cats = tx
    .select({ id: categories.id, name: categories.name })
    .from(categories)
    .where(eq(categories.householdId, householdId))
    .all() as { id: string; name: string }[];
  const catMap = Object.fromEntries(cats.map((c) => [c.name, c.id]));

  const merchantRows = tx
    .select({ id: merchants.id, name: merchants.name })
    .from(merchants)
    .where(eq(merchants.householdId, householdId))
    .all() as { id: string; name: string }[];
  const merchantMap = Object.fromEntries(merchantRows.map((m) => [m.name, m.id]));

  const SOURCES = ["rule", "rule", "rule", "pfc", "manual"] as const;
  let txnIndex = 0;

  function addTxn(opts: {
    accountId: string;
    date: string;
    name: string;
    amount: number;
    merchantName?: string;
    categoryName?: string;
    isIncome?: boolean;
  }) {
    const normalizedAmount = opts.isIncome ? Math.abs(opts.amount) : -Math.abs(opts.amount);
    const source = SOURCES[txnIndex % SOURCES.length];
    tx.insert(transactions).values({
      id: `demo-txn-${String(txnIndex).padStart(4, "0")}`,
      accountId: opts.accountId,
      householdId,
      date: opts.date,
      originalName: opts.name,
      name: opts.name,
      amount: opts.amount,
      normalizedAmount,
      merchantId: opts.merchantName ? merchantMap[opts.merchantName] ?? null : null,
      categoryId: opts.categoryName ? catMap[opts.categoryName] ?? null : null,
      categorySource: source,
      reviewed: txnIndex % 4 !== 0,
    }).run();
    txnIndex++;
  }

  // Generate 6 months of transactions
  for (let monthOffset = 0; monthOffset < 6; monthOffset++) {
    const baseDay = monthOffset * 30;

    // Salary (bi-weekly, 2x per month) — income
    addTxn({ accountId: DEMO_ACCOUNTS.checking, date: daysAgo(baseDay + 1), name: "ACME Corp Payroll", amount: -550000, merchantName: "ACME Corp", categoryName: "Salary", isIncome: true });
    addTxn({ accountId: DEMO_ACCOUNTS.checking, date: daysAgo(baseDay + 15), name: "ACME Corp Payroll", amount: -550000, merchantName: "ACME Corp", categoryName: "Salary", isIncome: true });

    // Rent
    addTxn({ accountId: DEMO_ACCOUNTS.checking, date: daysAgo(baseDay + 1), name: "Landlord Property Mgmt", amount: 180000, merchantName: "Landlord Property Mgmt", categoryName: "Rent/Mortgage" });

    // Subscriptions
    addTxn({ accountId: DEMO_ACCOUNTS.credit, date: daysAgo(baseDay + 5), name: "Netflix", amount: 1599, merchantName: "Netflix", categoryName: "Subscriptions" });
    addTxn({ accountId: DEMO_ACCOUNTS.credit, date: daysAgo(baseDay + 5), name: "Spotify", amount: 1099, merchantName: "Spotify", categoryName: "Subscriptions" });
    addTxn({ accountId: DEMO_ACCOUNTS.credit, date: daysAgo(baseDay + 3), name: "Planet Fitness", amount: 4500, merchantName: "Planet Fitness", categoryName: "Fitness" });

    // Utilities
    addTxn({ accountId: DEMO_ACCOUNTS.checking, date: daysAgo(baseDay + 10), name: "PG&E Electric", amount: 8500 + (monthOffset * 700), merchantName: "PG&E", categoryName: "Electric" });
    addTxn({ accountId: DEMO_ACCOUNTS.checking, date: daysAgo(baseDay + 12), name: "Comcast Internet", amount: 7999, merchantName: "Comcast", categoryName: "Internet" });
    addTxn({ accountId: DEMO_ACCOUNTS.checking, date: daysAgo(baseDay + 12), name: "T-Mobile", amount: 5500, merchantName: "T-Mobile", categoryName: "Phone" });

    // Car loan
    addTxn({ accountId: DEMO_ACCOUNTS.checking, date: daysAgo(baseDay + 15), name: "Auto Loan Payment", amount: 38500, categoryName: "Car Payment" });

    // Groceries (3x per month-ish)
    const groceryStores = ["Whole Foods", "Trader Joe's", "Safeway"];
    for (let g = 0; g < 3; g++) {
      const store = groceryStores[g % 3];
      const amount = 4500 + ((monthOffset * 3 + g) * 1100) % 7500;
      addTxn({ accountId: DEMO_ACCOUNTS.credit, date: daysAgo(baseDay + 4 + g * 8), name: store, amount, merchantName: store, categoryName: "Groceries" });
    }

    // Additional grocery trips (variable)
    for (let g = 0; g < 5; g++) {
      const store = groceryStores[(monthOffset + g) % 3];
      const amount = 2800 + ((monthOffset * 5 + g) * 1300) % 9000;
      addTxn({ accountId: DEMO_ACCOUNTS.credit, date: daysAgo(baseDay + 2 + g * 5), name: store, amount, merchantName: store, categoryName: "Groceries" });
    }

    // Restaurants (3-4x per month)
    const restaurants = ["Chipotle", "Sushi Zen"];
    for (let r = 0; r < 4; r++) {
      const rest = restaurants[r % 2];
      const amount = 1500 + ((monthOffset * 4 + r) * 900) % 6500;
      addTxn({ accountId: DEMO_ACCOUNTS.credit, date: daysAgo(baseDay + 6 + r * 6), name: rest, amount, merchantName: rest, categoryName: "Restaurants" });
    }

    // Coffee (4-5x per month)
    for (let c = 0; c < 5; c++) {
      const amount = 525 + ((monthOffset * 5 + c) * 75) % 300;
      addTxn({ accountId: DEMO_ACCOUNTS.credit, date: daysAgo(baseDay + 1 + c * 5), name: "Blue Bottle Coffee", amount, merchantName: "Blue Bottle Coffee", categoryName: "Coffee Shops" });
    }

    // Gas (2x per month)
    const gasStations = ["Shell", "Chevron"];
    for (let g = 0; g < 2; g++) {
      const station = gasStations[g];
      const amount = 4000 + ((monthOffset * 2 + g) * 500) % 2500;
      addTxn({ accountId: DEMO_ACCOUNTS.credit, date: daysAgo(baseDay + 7 + g * 12), name: station, amount, merchantName: station, categoryName: "Gas" });
    }

    // Shopping (sporadic)
    if (monthOffset % 2 === 0) {
      addTxn({ accountId: DEMO_ACCOUNTS.credit, date: daysAgo(baseDay + 18), name: "Amazon", amount: 3499 + monthOffset * 1200, merchantName: "Amazon", categoryName: "Electronics" });
    }
    if (monthOffset % 3 === 0) {
      addTxn({ accountId: DEMO_ACCOUNTS.credit, date: daysAgo(baseDay + 22), name: "Target", amount: 6700 + monthOffset * 800, merchantName: "Target", categoryName: "Home Goods" });
      addTxn({ accountId: DEMO_ACCOUNTS.credit, date: daysAgo(baseDay + 25), name: "Nordstrom", amount: 12500, merchantName: "Nordstrom", categoryName: "Clothing" });
    }
  }
}

function seedDemoBalanceHistory(tx: any) {
  // 180 days of daily balance snapshots for each account
  const accountBalances: Record<string, { base: number; variance: number }> = {
    [DEMO_ACCOUNTS.checking]: { base: 420000, variance: 50000 },
    [DEMO_ACCOUNTS.savings]: { base: 1250000, variance: 5000 },
    [DEMO_ACCOUNTS.credit]: { base: -180000, variance: 30000 },
    [DEMO_ACCOUNTS.investment]: { base: 4500000, variance: 100000 },
    [DEMO_ACCOUNTS.loan]: { base: -820000, variance: 3200 },
  };

  for (const [accountId, { base, variance }] of Object.entries(accountBalances)) {
    for (let day = 180; day >= 0; day--) {
      // Deterministic "variation" based on day index
      const offset = Math.round(Math.sin(day * 0.1) * variance * (1 - day / 360));
      const balance = base + offset;
      tx.insert(balanceHistory).values({
        id: uuid(),
        accountId,
        date: daysAgo(day),
        balance,
      }).run();
    }
  }
}
